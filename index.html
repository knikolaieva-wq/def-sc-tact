        // ---------- NFT Manager ----------
        let NFT_OPS = { mint: 0, burn: 0, setWl: 0, setProc: 0, incSuc: 0 };
        const nftStatus = document.getElementById('nftStatus');
        function logNft(msg) { nftStatus.textContent += (msg + '\n'); }
        function nftAddr() { return (document.getElementById('nftAddress').value||'').trim(); }
        async function readNftOps() {
          const addr = nftAddr(); if (!addr) throw new Error('Укажите адрес DefNFT');
          const client = await rpcClientForCurrentNet();
          const res = await client.runMethod(TonAddress.parse(addr), 'getOps');
          NFT_OPS.mint = Number(res.stack.readNumber());
          NFT_OPS.burn = Number(res.stack.readNumber());
          NFT_OPS.setWl = Number(res.stack.readNumber());
          NFT_OPS.setProc = Number(res.stack.readNumber());
          NFT_OPS.incSuc = Number(res.stack.readNumber());
          logNft(`NFT OPs: mint=${NFT_OPS.mint}, burn=${NFT_OPS.burn}, setWl=${NFT_OPS.setWl}, setProc=${NFT_OPS.setProc}, incSuc=${NFT_OPS.incSuc}`);
        }
        document.getElementById('nftReadOpsBtn').addEventListener('click', () =>
          readNftOps().catch(e => logNft('getOps error: ' + (e?.message||e)))
        );

        function b64(cell) { return cell.toBoc({ idx: false }).toString('base64'); }
        function storeStringToCell(str) {
          const enc = new TextEncoder();
          const bytes = enc.encode(str);
          const b = beginCell();
          for (const by of bytes) b.storeUint(by, 8);
          return b.endCell();
        }
        async function sendNftMessage({ to, payloadBase64, value = '0.05' }) {
          const tx = {
            validUntil: Math.floor(Date.now()/1000) + 300,
            messages: [{ address: to, amount: toNano(value).toString(), payload: payloadBase64 }]
          };
          logNft('Отправляем транзакцию…');
          await tonConnectUI.sendTransaction(tx);
          logNft('Транзакция отправлена');
        }

        // Mint
        document.getElementById('nftMintBtn').addEventListener('click', async () => {
          try {
            if (!NFT_OPS.mint) await readNftOps();
            const to = nftAddr(); if (!to) throw new Error('Укажите адрес DefNFT');
            const rcpt = document.getElementById('nftMintTo').value.trim(); if (!rcpt) throw new Error('Получатель');
            const bps = Number(document.getElementById('nftMintBps').value||'0'); if (bps<0||bps>10000) throw new Error('shareBps 0..10000');
            const uri = (document.getElementById('nftMintURI').value||'').trim(); if (!uri) throw new Error('tokenURI');
            const b = beginCell();
            b.storeUint(NFT_OPS.mint, 32);
            b.storeAddress(Address.parse(rcpt));
            b.storeInt(BigInt(bps), 257);
            const u = storeStringToCell(uri);
            b.storeRef(u);
            await sendNftMessage({ to, payloadBase64: b64(b.endCell()) });
          } catch (e) { logNft('Mint error: ' + (e?.message||e)); }
        });

        // Burn
        document.getElementById('nftBurnBtn').addEventListener('click', async () => {
          try {
            if (!NFT_OPS.burn) await readNftOps();
            const to = nftAddr(); if (!to) throw new Error('Укажите адрес DefNFT');
            const tid = BigInt(document.getElementById('nftBurnTokenId').value||'0'); if (tid<=0n) throw new Error('tokenId>0');
            const b = beginCell(); b.storeUint(NFT_OPS.burn, 32); b.storeInt(tid, 257);
            await sendNftMessage({ to, payloadBase64: b64(b.endCell()) });
          } catch (e) { logNft('Burn error: ' + (e?.message||e)); }
        });

        // Whitelist
        document.getElementById('nftWLBtn').addEventListener('click', async () => {
          try {
            if (!NFT_OPS.setWl) await readNftOps();
            const to = nftAddr(); if (!to) throw new Error('Укажите адрес DefNFT');
            const w = document.getElementById('nftWLWallet').value.trim(); if (!w) throw new Error('wallet');
            const allowed = document.getElementById('nftWLAllowed').value === 'true';
            const b = beginCell(); b.storeUint(NFT_OPS.setWl, 32); b.storeAddress(Address.parse(w)); b.storeBit(allowed);
            await sendNftMessage({ to, payloadBase64: b64(b.endCell()) });
          } catch (e) { logNft('Whitelist error: ' + (e?.message||e)); }
        });

        // Processor
        document.getElementById('nftProcBtn').addEventListener('click', async () => {
          try {

            const allowed = document.getElementById('nftProcAllowed').value === 'true';
            const b = beginCell(); b.storeUint(NFT_OPS.setProc, 32); b.storeAddress(Address.parse(p)); b.storeBit(allowed);
            await sendNftMessage({ to, payloadBase64: b64(b.endCell()) });
          } catch (e) { logNft('Processor error: ' + (e?.message||e)); }
        });

        // Increment successful tx

          try {

            const w = document.getElementById('nftIncWallet').value.trim(); if (!w) throw new Error('wallet');
            const b = beginCell(); b.storeUint(NFT_OPS.incSuc, 32); b.storeAddress(Address.parse(w));
            await sendNftMessage({ to, payloadBase64: b64(b.endCell()), value: '0.05' });
          } catch (e) { logNft('Increment error: ' + (e?.message||e)); }
        });

        // Getters
        document.getElementById('nftTokenOfBtn').addEventListener('click', async () => {
          const out = document.getElementById('nftTokenOfOut'); out.textContent='';
          try { const addr = nftAddr(); if (!addr) throw new Error('Укажите адрес DefNFT');
            const who = document.getElementById('nftGetOwner').value.trim(); if (!who) throw new Error('owner');
            const client = await rpcClientForCurrentNet();
            const res = await client.runMethod(TonAddress.parse(addr), 'tokenOf', [{ type:'slice', cell: beginCell().storeAddress(Address.parse(who)).endCell() }]);
            out.textContent = String(res.stack.readNumber());
          } catch (e) { out.textContent = 'tokenOf error: ' + (e?.message||e); }
        });
